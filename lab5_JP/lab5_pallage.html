Question 1: Transformer un ADNLPModel en un problme pnalis

Coder la fonction quad_penalty_adnlp qui prend en entre un ADNLPModel, et un paramtre et qui retourne un nouveau ADNLPModel qui correspond au problme sans contrainte: $$ \min_x f(x) + \frac{\rho}{2}\|c(x)\|^2. $$ Remarque: on peut accder aux fonctions f et c par NLPModels.obj() et NLPModels.cons().

In[]:
using NLPModels, NLPModelsIpopt



function quad_penalty_adnlp(nlp :: ADNLPModel,  :: Real)
    """
    Returns a new NLPModel with a quadratic penalty term added to the objective function.
    """
    f = x -> obj(nlp,x) + /2 .* norm(cons(nlp,x))^2
    nlp_quad = ADNLPModel(f, nlp.meta.x0)
   return nlp_quad
end

function cv_model(n :: Int)

  domain = (0,1) # set the domain
  partition = n
  model = CartesianDiscreteModel(domain,partition) # set discretization
    
  labels = get_face_labeling(model)
  add_tag_from_tags!(labels,"diri1",[2])
  add_tag_from_tags!(labels,"diri0",[1]) # boundary conditions

  order=1
  valuetype=Float64
  reffe = ReferenceFE(lagrangian, valuetype, order)
  V0 = TestFESpace(model, reffe; conformity=:H1, dirichlet_tags=["diri0","diri1"])
  U = TrialFESpace(V0,[0., exp(1)-exp(-2)])

  trian = Triangulation(model)
  degree = 2
  d = Measure(trian,degree) # integration machinery

  # Our objective function
  w(x) = exp(x[1])
  function f(y)
    (((y)(y) + 2 * y * y) * w) * d
  end

  xin = zeros(Gridap.FESpaces.num_free_dofs(U))
  nlp = GridapPDENLPModel(xin, f, trian, U, V0)
  return nlp
end
