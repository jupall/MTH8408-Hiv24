"""
Function that implements the Gauss-Newton algorithm for Nonlinear Least-square resolution.
"""
function gauss_newton(nlp      :: AbstractNLSModel, 
                      x        :: AbstractVector, 
                              :: AbstractFloat = 1e-6;
                             :: AbstractFloat = 1e-3, 
                             :: AbstractFloat = 0.66, 
                             :: AbstractFloat = 0.25, 
                             :: AbstractFloat = 2.0,
                      max_eval :: Int = 1_000, 
                      max_time :: AbstractFloat = 60.,
                      max_iter :: Int = typemax(Int64)
                      )
    ######################################################
    Fx = residual(nlp, x)# le rsidu
    Jx =  jac_residual(nlp, x) # operateur qui reprsente le jacobien du rsidu
    ######################################################
    normFx = norm(Fx)

     = 1.

    iter = 0    

    el_time = 0.0
    tired   = neval_residual(nlp) > max_eval || el_time > max_time
    status  = :unknown

    start_time = time()
    too_small  = false
    normdual   = norm(Jx' * Fx)
    optimal    = min(normFx, normdual)  

    @info log_header([:iter, :nf, :primal, :status, :nd, :],
    [Int, Int, Float64, String, Float64, Float64],
    hdr_override=Dict(:nf => "#F", :primal => "F(x)", :nd => "d"))

    while !(optimal || tired || too_small)

        #################################
        #Compute a direction satisfying the trust-region constraint
        (d, stats)  = lsmr(-Jx,Fx,radius=)
        #################################
      
        too_small = norm(d) < 1e-15
        if too_small #the direction is too small
            status = :too_small
        else
            xp      = x + d
            ###########################
            Fxp     =  residual(nlp, xp)# value le rsidu en xp
            ###########################
            normFxp = norm(Fxp)

            Pred = 0.5 * (normFx^2 - norm(Jx * d + Fx)^2)
            Ared = 0.5 * (normFx^2 - normFxp^2)

            if Ared/Pred < 
                 = max(1e-8,  * )
                status = :reduce_
            else #success
                x  = xp
                ###########################
                Jx = jac_residual(nlp, x)# revalue le jacobien en x
                ###########################
                Fx = Fxp
                normFx = normFxp
                status = :success
                if Ared/Pred >  && norm(d) >= 0.99 * 
                     *= 
                end
            end
        end

        @info log_row(Any[iter, neval_residual(nlp), normFx, status, norm(d), ])

        el_time      = time() - start_time
        iter   += 1

        many_evals   = neval_residual(nlp) > max_eval
        iter_limit   = iter > max_iter
        tired        = many_evals || el_time > max_time || iter_limit
        normdual     = norm(Jx' * Fx)
        optimal      = min(normFx, normdual)  
    end

    status = if optimal 
        :first_order
    elseif tired
        if neval_residual(nlp) > max_eval
            :max_eval
        elseif el_time > max_time
            :max_time
        elseif iter > max_iter
            :max_iter
        else
            :unknown_tired
        end
    elseif too_small
        :stalled
    else
        :unknown
    end

    return GenericExecutionStats(nlp; status, solution = x,
                                 objective = normFx^2 / 2,
                                 dual_feas = normdual,
                                 iter = iter, 
                                 elapsed_time = el_time)
end

Exercice 2: Mthode Levenberg-Marquard inexacte

Dans cet exercice, on complte une implmentation de la mthode Levenberg-Marquardt. Pour complter le code lm_param on va utiliser les fonctions suivantes:

  • dsol qui calcul la solution du systme $\min_x \frac{1}{2}\|J(x) d + F(x)\| + \lambda \|x\|^2$ avec la fonction lsqr du package Krylov.jl.
  • multi_sol qui pour un entier nl donn et un $\mu$ va rsoudre le problme de dsol pour nl valeurs de $\lambda$ (autour de la valeur $\mu$). Par exemple, pour $\mu=10^{-6}$ et $nl=3$, on prendra $\lambda=10^{-7}, 10^{-6},10^{-5}$. Parmis les nl directions calcules, on retourne celle qui donne la plus petite valeur de $\|F(x+d)\|^2$.

"""
Function that generate a list of lambdas according to the requirements.
Inputs: mean lambda, number of lambda to generate 
Output: List of lambdas
"""
function generate_lambdas( :: Float64, nl :: Int64)
    nl = (mod(nl, 2) == 0) ? nl + 1 : nl + 0     
    lambda_list = []
    push!(lambda_list, )
    half = trunc(nl/2)
    for i in range(start = 1, stop = half, step = 1)
        plus_lamb = ( * 10^(i))
        minus_lamb = ( * 10^(-i))
        append!(lambda_list, plus_lamb)
        append!(lambda_list, minus_lamb)
    end
    return lambda_list
   
end


"""
Function that generate the best direction according to different regularization parameters.
Inputs: nlp, x, Fx, Jx, lambda, Tau
Output: best direction
"""
function multi_sol(nlp, x, Fx, Jx, , ; nl = 3)
    lambda_list = generate_lambdas(, Int(nl))
    count = 0
    best_NormFx = 0
    best_d = 0
    for lambd in lambda_list
        (d, stats)=  dsol(Fx, Jx, lambd, )
        nextX = x + d
        nextFx = residual(nlp, nextX)
        next_normFx = (norm(nextFx)).^2
        if count == 0
            best_NormFx = next_normFx
            best_d = d
        elseif (next_normFx < best_NormFx)
            best_NormFx = next_normFx
            best_d = d
        else
            continue 
        end
        
        count += 1
    end
    return best_d
end

using JuMP, Ipopt

# Create JuMP model, using Ipopt as the solver
rocket = Model(optimizer_with_attributes(Ipopt.Optimizer, "print_level" => 0))

# Constants
# Note that all parameters in the model have been normalized
# to be dimensionless. See the COPS3 paper for more info.
h_0 = 1    # Initial height
v_0 = 0    # Initial velocity
m_0 = 1    # Initial mass
g_0 = 1    # Gravity at the surface

T_c = 3.5  # Used for thrust
h_c = 500  # Used for drag
v_c = 620  # Used for drag
m_c = 0.6  # Fraction of initial mass left at end

c     = 0.5 * sqrt(g_0 * h_0)  # Thrust-to-fuel mass
m_f   = m_c * m_0            # Final mass
D_c   = 0.5 * v_c * m_0 / g_0    # Drag scaling
T_max = T_c * g_0 * m_0        # Maximum thrust

n = 800   # Time steps

@variables(rocket, begin
    t  0, (start = 1/n) # Time step
    # State variables
    v[1:n]  0            # Velocity
    h[1:n]  h_0          # Height
    m_f  m[1:n]  m_0    # Mass
    # Control
    0  T[1:n]  T_max    # Thrust
end)

# Objective: maximize altitude at end of time of flight
@objective(rocket, Max, h[n])

# Initial conditions
@constraints(rocket, begin
    v[1] == v_0
    h[1] == h_0
    m[1] == m_0
    m[n] == m_f
end)

# Forces
# Drag(h,v) = Dc v^2 exp( -hc * (h - h0) / h0 )
@NLexpression(rocket, drag[j = 1:n], D_c * (v[j]^2) * exp(-h_c * (h[j] - h_0) / h_0))
# Grav(h)   = go * (h0 / h)^2
@NLexpression(rocket, grav[j = 1:n], g_0 * (h_0 / h[j])^2)
# Time of flight
@NLexpression(rocket, t_f, t * n)

# Dynamics
for j in 2:n
    # h' = v
    
    # Rectangular integration
    # @NLconstraint(rocket, h[j] == h[j - 1] + t * v[j - 1])
    
    # Trapezoidal integration
    @NLconstraint(rocket,
        h[j] == h[j - 1] + 0.5 * t * (v[j] + v[j - 1]))

    # v' = (T-D(h,v))/m - g(h)
    
    # Rectangular integration
    # @NLconstraint(rocket, v[j] == v[j - 1] + t *(
    #                 (T[j - 1] - drag[j - 1]) / m[j - 1] - grav[j - 1]))
    
    # Trapezoidal integration
    @NLconstraint(rocket,
        v[j] == v[j-1] + 0.5 * t * (
            (T[j] - drag[j] - m[j] * grav[j]) / m[j] +
            (T[j - 1] - drag[j - 1] - m[j - 1] * grav[j - 1]) / m[j - 1]))

    # m' = -T/c

    # Rectangular integration
    # @NLconstraint(rocket, m[j] == m[j - 1] - t * T[j - 1] / c)
    
    # Trapezoidal integration
    @NLconstraint(rocket,
        m[j] == m[j - 1] - 0.5 * t * (T[j] + T[j-1]) / c)
end
